def distinct_label_count(Z_basis, Q, h):
    # Exact braid-arrangement inclusion-exclusion.
    total = 0

    for partition in set_partitions(range(h)):
        constraints = []
        for block in partition:
            root = block[0]
            for i in block[1:]:
                constraints.append(z_coordinate(i) - z_coordinate(root))

        dim_intersection = dimension_of_subspace_with_constraints(
            Z_basis, constraints
        )

        mu = 1
        for block in partition:
            s = len(block)
            mu *= (-1)**(s - 1) * factorial(s - 1)

        total += mu * Q**dim_intersection

    return total
